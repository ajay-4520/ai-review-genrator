async function fetchUserLogins() { try { const response = await fetch("http://localhost:3000/api/admin/signed-users"); const users = await response.json(); const tbody = document.getElementById("userTableBody"); tbody.innerHTML = ""; users.forEach(user => { const row = document.createElement("tr"); row.innerHTML = ` ${user.email} ${user.method} ${new Date(user.time).toLocaleString()} `; tbody.appendChild(row); }); } catch (err) { alert("⚠️ Failed to fetch login data."); console.error(err); } }
